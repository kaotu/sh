#./bin/bash

# Install PHP&PHP-FPM 
sudo rpm -Uvh https://mirror.webtatic.com/yum/el7/webtatic-release.rpm
sudo yum install -y php71w php71w-curl php71w-common php71w-cli php71w-mysql php71w-mbstring php71w-fpm php71w-xml php71w-pdo php71w-zip

#File configuration

sudo nano /etc/php.ini
#>> Fix and Uncomment 
	cgi.fix_pathinfo=0

sodo nano /etc/php-fpm.d/www.conf
# >> Fix and uncomment <<
	user = nginx
	group = nginx
	listen = /run/php-fpm/php-fpm.sock
	listen.owner = nginx
	listen.group = nginx
	listen.mode  = 0660
	env[HOSTNAME] = $HOSTNAME
	env[PATH] = /usr/local/bin:/usr/bin:/bin
	env[TMP] = /tmp
	env[TMPDIR] = /tmp
	env[TEMP] = /tmp
sudo systemctl start php-fpm
sudo systemctl enable php-fpm

#Check running
sudo netstat -pl | grep php-fpm.sock

#Install MariaDB Server#
sudo yum -y install mariadb mariadb-server
sudo systemctl start mariadb
sudo systemctl enable mariadb
#Configuration of root password for MariaDB
sudo mysql_secure_installation #all is "yes"

#Install PHP Composer#
sudo curl -sS https://getcomposer.org/installer | sudo php -- --install-dir=/usr/bin --filename=composer
sudo composer
